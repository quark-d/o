Public Result As c_LedgerDialogResult
Private mCtx As c_DialogContext

Private Sub UserForm_Initialize()
    Set Result = New c_LedgerDialogResult
    Result.Init "cancel", Nothing, ""
End Sub

Public Sub InitDialog(ByVal ctx As c_DialogContext)
    Set mCtx = ctx
    Me.Caption = "”ہ“ü (‘qŒةA پ¨ ‘qŒةB)"

    Dim it As c_ItemInfo
    For Each it In ctx.Items
        cmbItem.AddItem it.ItemNo & "  " & it.ItemName
    Next
    m_LedgerFormUtil.FillCommon mCtx, cmbRecorder, txtDate
End Sub

Private Sub cmbItem_Change()
    If cmbItem.ListIndex < 0 Then Exit Sub
    Dim it As c_ItemInfo: Set it = mCtx.Items(cmbItem.ListIndex + 1)
    lblPool.Caption = "Œ»چف‚ج–¢ٹ„“–چفŒة: " & _
        m_LedgerFormUtil.DescribeStates(mCtx, "U", it.ItemNo)
End Sub

Private Sub btnOK_Click()
    If cmbItem.ListIndex < 0 Then
        MsgBox "•i”ش‚ً‘I‘ً‚µ‚ؤ‚­‚¾‚³‚¢پB", vbExclamation: Exit Sub
    End If
    If Not m_LedgerFormUtil.CheckCommon(txtQty, txtDate, cmbRecorder) Then Exit Sub

    Dim op As New c_LedgerOperation
    op.Kind = LedgerOpKind.opCarryIn
    op.ItemNo = mCtx.Items(cmbItem.ListIndex + 1).ItemNo
    op.Qty = CLng(txtQty.Text)
    op.ActionDate = CDate(txtDate.Text)
    op.Recorder = cmbRecorder.Text
    op.Note = Trim$(txtNote.Text)
    Result.Init "ok", op, ""
    Me.Hide
End Sub

Private Sub btnCancel_Click()
    Me.Hide
End Sub

Private Sub UserForm_QueryClose(Cancel As Integer, CloseMode As Integer)
    If CloseMode = vbFormControlMenu Then
        Cancel = True
        Me.Hide
    End If
End Sub
