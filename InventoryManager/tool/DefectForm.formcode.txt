Option Explicit

'==============================================================================
' 不良+補填ダイアログ (不良発生 1 行 + 補填 ON なら充当 1 行を自動生成)
'
' コントロールは build_xlsm.ps1 がデザイン時に生成する:
'   cmbLot / lblItem / cmbState / lblStock / txtQty / chkRefill /
'   cmbRefillState / txtRefillQty / lblPool / txtDate / cmbRecorder /
'   txtNote / btnOK / btnCancel
'==============================================================================

Public Result As c_LedgerDialogResult
Private mCtx As c_DialogContext
Private mLots As Collection        ' 有効ロットのみ (cmbLot と同順)

Private Sub UserForm_Initialize()
    Set Result = New c_LedgerDialogResult
    Result.Init "cancel", Nothing, ""
End Sub

Public Sub InitDialog(ByVal ctx As c_DialogContext)
    Set mCtx = ctx
    Me.Caption = "不良発生 (+補填)"

    Set mLots = New Collection
    Dim lt As c_LotInfo
    Dim it As c_ItemInfo
    For Each lt In ctx.Lots
        If lt.PlanStatus = "有効" Then
            mLots.Add lt
            Set it = ctx.FindItemByLocalId(lt.LocalId)
            cmbLot.AddItem lt.LotId & _
                IIf(it Is Nothing, "", "  " & it.ItemName) & "  (必要 " & lt.Required & ")"
        End If
    Next
    SetRefillEnabled False
    m_LedgerFormUtil.FillCommon mCtx, cmbRecorder, txtDate
End Sub

Private Sub cmbLot_Change()
    cmbState.Clear
    lblStock.Caption = ""
    cmbRefillState.Clear
    lblPool.Caption = ""
    If cmbLot.ListIndex < 0 Then Exit Sub

    Dim lt As c_LotInfo: Set lt = mLots(cmbLot.ListIndex + 1)
    Dim it As c_ItemInfo: Set it = mCtx.FindItemByLocalId(lt.LocalId)
    If Not it Is Nothing Then
        lblItem.Caption = m_LedgerFormUtil.DescribeItem(it)
    Else
        lblItem.Caption = lt.LocalId
    End If

    ' 不良の元状態 = そのロットに在庫がある状態のみ
    m_LedgerFormUtil.FillCombo cmbState, mCtx.Stock.StatesWithStock("L", lt.LotId)
    ' 補填元 = 同品番 (localId) の未割当在庫がある状態のみ
    m_LedgerFormUtil.FillCombo cmbRefillState, mCtx.Stock.StatesWithStock("U", lt.LocalId)
    lblPool.Caption = "未割当在庫: " & m_LedgerFormUtil.DescribeStates(mCtx, "U", lt.LocalId)
End Sub

Private Sub cmbState_Change()
    If cmbLot.ListIndex < 0 Or cmbState.ListIndex < 0 Then Exit Sub
    Dim lt As c_LotInfo: Set lt = mLots(cmbLot.ListIndex + 1)
    lblStock.Caption = "可用 " & _
        mCtx.Stock.AvailableQty("L", lt.LotId, cmbState.Text) & " 個"
End Sub

Private Sub chkRefill_Click()
    SetRefillEnabled chkRefill.Value
    If chkRefill.Value Then SyncRefillQty
End Sub

Private Sub txtQty_Change()
    If chkRefill.Value Then SyncRefillQty
End Sub

Private Sub cmbRefillState_Change()
    If chkRefill.Value Then SyncRefillQty
End Sub

' 補填数量の既定 = MIN(不良数量, 選択した補填元状態の未割当在庫)
Private Sub SyncRefillQty()
    If cmbLot.ListIndex < 0 Or cmbRefillState.ListIndex < 0 Then Exit Sub
    If Not IsNumeric(txtQty.Text) Then Exit Sub
    Dim lt As c_LotInfo: Set lt = mLots(cmbLot.ListIndex + 1)
    Dim pool As Long: pool = mCtx.Stock.AvailableQty("U", lt.LocalId, cmbRefillState.Text)
    Dim q As Long: q = CLng(txtQty.Text)
    If pool < q Then q = pool
    txtRefillQty.Text = CStr(q)
End Sub

Private Sub SetRefillEnabled(ByVal enabled As Boolean)
    cmbRefillState.Enabled = enabled
    txtRefillQty.Enabled = enabled
End Sub

Private Sub btnOK_Click()
    If cmbLot.ListIndex < 0 Then
        MsgBox "ロットを選択してください。", vbExclamation: Exit Sub
    End If
    If cmbState.ListIndex < 0 Then
        MsgBox "不良になった在庫の元状態を選択してください。", vbExclamation: Exit Sub
    End If
    If Not m_LedgerFormUtil.CheckCommon(txtQty, txtDate, cmbRecorder) Then Exit Sub

    Dim op As New c_LedgerOperation
    op.Kind = LedgerOpKind.opDefect
    op.LotId = mLots(cmbLot.ListIndex + 1).LotId
    op.FromState = cmbState.Text
    op.Qty = CLng(txtQty.Text)
    op.ActionDate = CDate(txtDate.Text)
    op.Recorder = cmbRecorder.Text
    op.Note = Trim$(txtNote.Text)

    If chkRefill.Value Then
        If cmbRefillState.ListIndex < 0 Then
            MsgBox "補填元の状態を選択してください。", vbExclamation: Exit Sub
        End If
        If Not IsNumeric(txtRefillQty.Text) Or CLng(Val(txtRefillQty.Text)) < 1 Then
            MsgBox "補填数量を 1 以上の整数で入力してください。", vbExclamation: Exit Sub
        End If
        op.Refill = True
        op.RefillFromState = cmbRefillState.Text
        op.RefillQty = CLng(txtRefillQty.Text)
    End If

    Result.Init "ok", op, ""
    Me.Hide
End Sub

Private Sub btnCancel_Click()
    Me.Hide
End Sub

Private Sub UserForm_QueryClose(Cancel As Integer, CloseMode As Integer)
    If CloseMode = vbFormControlMenu Then
        Cancel = True
        Me.Hide
    End If
End Sub
