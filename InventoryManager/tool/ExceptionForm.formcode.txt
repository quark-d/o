Option Explicit

'==============================================================================
' 例外操作ダイアログ (廃棄 / 倉庫戻し / 余剰化 / 充当)
'
' 頻度の低い操作を 1 画面に集約。対象は「ロット」または「未割当 (品番)」。
' コントロールは build_xlsm.ps1 がデザイン時に生成する:
'   optDiscard / optReturn / optSurplus / optAllocate / cmbTarget / cmbState /
'   lblStock / txtQty / cmbTargetLot / txtDate / cmbRecorder / txtNote /
'   btnOK / btnCancel
'==============================================================================

Public Result As c_LedgerDialogResult
Private mCtx As c_DialogContext
Private mTargetLotIds As Collection   ' cmbTarget と同順 (ロットID。未割当は "")
Private mTargetLocals As Collection   ' cmbTarget と同順 (localId)
Private mTargetLots As Collection     ' cmbTargetLot と同順 (充当先候補の c_LotInfo)

Private Sub UserForm_Initialize()
    Set Result = New c_LedgerDialogResult
    Result.Init "cancel", Nothing, ""
End Sub

Public Sub InitDialog(ByVal ctx As c_DialogContext)
    Set mCtx = ctx
    Me.Caption = "例外操作 (廃棄 / 倉庫戻し / 余剰化 / 充当)"
    optDiscard.Value = True
    m_LedgerFormUtil.FillCommon mCtx, cmbRecorder, txtDate
    RebuildTargets
End Sub

Private Sub optDiscard_Click()
    RebuildTargets
End Sub
Private Sub optReturn_Click()
    RebuildTargets
End Sub
Private Sub optSurplus_Click()
    RebuildTargets
End Sub
Private Sub optAllocate_Click()
    RebuildTargets
End Sub

' 選択中の例外操作イベント名
Private Function CurrentEvent() As String
    If optDiscard.Value Then CurrentEvent = m_LedgerCore.EV_DISCARD
    If optReturn.Value Then CurrentEvent = m_LedgerCore.EV_RETURN
    If optSurplus.Value Then CurrentEvent = m_LedgerCore.EV_SURPLUS
    If optAllocate.Value Then CurrentEvent = m_LedgerCore.EV_ALLOCATE
End Function

' 対象リストを操作種別に合わせて作り直す
'   廃棄     … 不良在庫があるロット/未割当
'   倉庫戻し … 何らかの在庫があるロット/未割当
'   余剰化   … 何らかの在庫があるロットのみ
'   充当     … 何らかの在庫があるロット/未割当 (充当元)
Private Sub RebuildTargets()
    Dim ev As String: ev = CurrentEvent()

    cmbTarget.Clear
    cmbState.Clear
    lblStock.Caption = ""
    Set mTargetLotIds = New Collection
    Set mTargetLocals = New Collection

    Dim lt As c_LotInfo
    For Each lt In mCtx.Lots
        If TargetHasStock(ev, "L", lt.LotId) Then
            mTargetLotIds.Add lt.LotId
            mTargetLocals.Add lt.LocalId
            cmbTarget.AddItem "ロット: " & lt.LotId & _
                IIf(lt.PlanStatus <> "有効", "  [" & lt.PlanStatus & "]", "")
        End If
    Next

    If ev <> m_LedgerCore.EV_SURPLUS Then
        Dim it As c_ItemInfo
        For Each it In mCtx.Items
            If TargetHasStock(ev, "U", it.LocalId) Then
                mTargetLotIds.Add ""
                mTargetLocals.Add it.LocalId
                cmbTarget.AddItem "未割当: " & m_LedgerFormUtil.DescribeItem(it)
            End If
        Next
    End If

    cmbTargetLot.Clear
    cmbTargetLot.Enabled = (ev = m_LedgerCore.EV_ALLOCATE)
    cmbState.Enabled = (ev <> m_LedgerCore.EV_DISCARD)
    If ev = m_LedgerCore.EV_DISCARD Then cmbState.AddItem m_LedgerCore.ST_DEFECT
End Sub

Private Function TargetHasStock(ByVal ev As String, ByVal bucket As String, _
                                ByVal id As String) As Boolean
    If ev = m_LedgerCore.EV_DISCARD Then
        TargetHasStock = (mCtx.Stock.AvailableQty(bucket, id, m_LedgerCore.ST_DEFECT) > 0)
    Else
        TargetHasStock = (mCtx.Stock.StatesWithStock(bucket, id).Count > 0)
    End If
End Function

Private Sub cmbTarget_Change()
    lblStock.Caption = ""
    If cmbTarget.ListIndex < 0 Then Exit Sub
    Dim idx As Long: idx = cmbTarget.ListIndex + 1

    Dim bucket As String, id As String
    If mTargetLotIds(idx) <> "" Then
        bucket = "L": id = mTargetLotIds(idx)
    Else
        bucket = "U": id = mTargetLocals(idx)
    End If

    If CurrentEvent() = m_LedgerCore.EV_DISCARD Then
        cmbState.Clear
        cmbState.AddItem m_LedgerCore.ST_DEFECT
        cmbState.ListIndex = 0
    Else
        m_LedgerFormUtil.FillCombo cmbState, mCtx.Stock.StatesWithStock(bucket, id)
    End If

    ' 充当先候補 = 同品番 (localId) の有効ロット (充当元自身は除く)
    If CurrentEvent() = m_LedgerCore.EV_ALLOCATE Then
        cmbTargetLot.Clear
        Set mTargetLots = New Collection
        Dim lt As c_LotInfo
        For Each lt In mCtx.Lots
            If lt.PlanStatus = "有効" And lt.LocalId = mTargetLocals(idx) _
               And lt.LotId <> mTargetLotIds(idx) Then
                mTargetLots.Add lt
                cmbTargetLot.AddItem lt.LotId & "  (必要 " & lt.Required & ")"
            End If
        Next
    End If
End Sub

Private Sub cmbState_Change()
    If cmbTarget.ListIndex < 0 Or cmbState.ListIndex < 0 Then Exit Sub
    Dim idx As Long: idx = cmbTarget.ListIndex + 1
    Dim avail As Long
    If mTargetLotIds(idx) <> "" Then
        avail = mCtx.Stock.AvailableQty("L", mTargetLotIds(idx), cmbState.Text)
    Else
        avail = mCtx.Stock.AvailableQty("U", mTargetLocals(idx), cmbState.Text)
    End If
    lblStock.Caption = "可用 " & avail & " 個"
End Sub

Private Sub btnOK_Click()
    If cmbTarget.ListIndex < 0 Then
        MsgBox "対象を選択してください。", vbExclamation: Exit Sub
    End If
    If cmbState.ListIndex < 0 Then
        MsgBox "元状態を選択してください。", vbExclamation: Exit Sub
    End If
    If CurrentEvent() = m_LedgerCore.EV_ALLOCATE And cmbTargetLot.ListIndex < 0 Then
        MsgBox "充当先ロットを選択してください。", vbExclamation: Exit Sub
    End If
    If Not m_LedgerFormUtil.CheckCommon(txtQty, txtDate, cmbRecorder) Then Exit Sub

    Dim idx As Long: idx = cmbTarget.ListIndex + 1

    Dim op As New c_LedgerOperation
    op.Kind = LedgerOpKind.opException
    op.EventKind = CurrentEvent()
    op.LotId = mTargetLotIds(idx)
    op.LocalId = mTargetLocals(idx)
    op.FromState = cmbState.Text
    op.Qty = CLng(txtQty.Text)
    op.ActionDate = CDate(txtDate.Text)
    op.Recorder = cmbRecorder.Text
    op.Note = Trim$(txtNote.Text)
    If op.EventKind = m_LedgerCore.EV_ALLOCATE Then
        op.TargetLotId = mTargetLots(cmbTargetLot.ListIndex + 1).LotId
    End If

    Result.Init "ok", op, ""
    Me.Hide
End Sub

Private Sub btnCancel_Click()
    Me.Hide
End Sub

Private Sub UserForm_QueryClose(Cancel As Integer, CloseMode As Integer)
    If CloseMode = vbFormControlMenu Then
        Cancel = True
        Me.Hide
    End If
End Sub
